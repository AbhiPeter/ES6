1)create person and student class using inhertinace:-

   class Person { 
    constructor(firstName: string, lastName: string) { 
        this.firstName = firstName;
        this.lastName = lastName;

    }
    getFullName() { 
        return this.firstName + ' ' + this.lastName;
    }
}

//here class student is the child class for person base class:-

class Student extends Person { 
    constructor(firstName: string, lastName: string) { 
        super(firstName, lastName);

    }
}

let studentFullName = new Person("peter", "parker");

console.log(studentFullName.getFullName());
