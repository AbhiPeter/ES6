class Person { 
    firstName: string;
    lastName: string;
    constructor(firstName, lastName)
    {
        this.firstName = firstName;
        this.lastName = lastName;
    };

    getFullName() { 
        return 'the name is given as ' + this.firstName + ' ' + this.lastName;

    };

    
};

let personFullName = new Person("peter", "parker");

console.log(personFullName.getFullName());
